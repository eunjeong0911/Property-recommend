Step 1: 데이터 전처리
# 순서대로 실행 (크롤링 데이터 + V-WORLD API 데이터 통합)
docker compose exec reco python -m trust_model.data_preprocessing._00_load_Landbroker
docker compose exec reco python -m trust_model.data_preprocessing._01_load_broker
docker compose exec reco python -m trust_model.data_preprocessing._02_load_brokerOffice
docker compose exec reco python -m trust_model.data_preprocessing._03_merge_all_brokers
docker compose exec reco python -m trust_model.data_preprocessing._04_clean_broker
docker compose exec reco python -m trust_model.data_preprocessing._05_group_by_office

# 또는 한번에 실행
docker compose exec reco python -m trust_model.data_preprocessing.run_all_preprocessing
Step 2: 모델 학습 (pkl 파일 생성)
# 전체 파이프라인 실행 (데이터 로드 → Target 생성 → Feature 생성 → 학습 → 평가 → 모델 저장)
docker compose exec reco python -m trust_model.pipeline.run_all
→ 결과: apps/reco/trust_model/model/final_trust_model.pkl 생성

Step 3: DB 적재 (예측 결과 저장)
# 학습된 모델로 모든 중개사 신뢰도 등급 예측 → DB 저장
docker compose --profile scripts run --rm scripts python 03_import/trust/predict_trust_scores.py
한번에 실행 (전체)
docker compose --profile scripts run --rm scripts python 03_import/trust/import_trust_all.py
2️⃣ 가격 평가 모델 (Price Classification)
Step 1: 데이터 전처리
# 월세 데이터셋 전처리 파일 생성
docker compose exec reco python -m models.price_model.prepare_wolse_dataset
Step 2: 모델 학습 (pkl 파일 생성)
# 모델 학습 (LightGBM)
docker compose exec reco python -m models.price_model.ML.src.main
→ 결과: apps/reco/models/price_model/ML/model/price_model_lightgbm.pkl 생성

Step 3: SHAP 분석 (선택)
docker compose exec reco python -m models.price_model.ML.src.example_shap
Step 4: DB 적재 (피터팬 매물에 모델 적용)
# 학습된 모델을 JSON 매물 데이터에 적용 → DB 저장
docker compose exec reco python -m models.price_model.ML.src.apply_model_to_json
